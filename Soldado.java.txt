import java.util.Random;

public class Soldado {
    private String reino;
    private int nivelVida;

    public Soldado(String reino) {
        this.reino = reino;
        this.nivelVida = new Random().nextInt(10) + 1; // Nivel de vida entre 1 y 10
    }

    public String getReino() {
        return reino;
    }

    public int getNivelVida() {
        return nivelVida;
    }

    public void aumentarVida() {
        this.nivelVida += 1;
    }

    public void recibirDanio(int danio) {
        this.nivelVida -= danio;
        if (this.nivelVida < 0) this.nivelVida = 0;
    }
    
    public String toString() {
        return "Soldado del reino " + reino + " con nivel de vida " + nivelVida;
    }
}
